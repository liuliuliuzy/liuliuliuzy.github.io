            return res
        
        def mergeAndCount(left: int, right: int) -> int:
            if left == right: return 0
            mid = (left + right) // 2
            res = mergeAndCount(left, mid) + mergeAndCount(mid + 1, right) + countPairs(left, mid, right)
            i,j,k = left, mid+1, left
